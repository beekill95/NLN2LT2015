class Shape {
	length, width:float;
	float getArea() {}
	Shape(length, width:float) {
		self.length := length;
		self.width := width;
	}
}
class Rectangle extends Shape {
	float getArea() {
		return self.length*self.width;
	}
}
class Triangle extends Shape {
	float getArea() {
		return self.length*self.width/2;
	}
}
class Example2 {
	void main() {
		s:Shape;
		s := new Rectangle(3,4);
		io.writeFloatLn(s.getArea());
		s := new Triangle(3,4);
		io.writeFloatLn(s.getArea());
	}
}
